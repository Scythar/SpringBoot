Observer Pattern:
Observer pattern defines a one-to-many relationship where multiple observers are notified automatically when the subject state changes. It helps achieve loose coupling between subject and observers.

⭐Observer Interface & Implementations:
// Observer package
public interface Observer {
    void update(String message);
}

public class EmailSubscriber implements Observer {
    String name;

    public EmailSubscriber(String name) {
        this.name = name;
    }

    @Override
    public void update(String message) {
        System.out.println(name + " : sending via Email : " + message);
    }
}

public class SmsSubscriber implements Observer {
    String name;

    public SmsSubscriber(String name) {
        this.name = name;
    }

    @Override
    public void update(String message) {
        System.out.println(name + " : sending via SMS : " + message);
    }
}

// Subject package
public interface Subject {
    void addObserver(Observer o);       // ✅ Observer param
    void removeObserver(Observer o);    // ✅ Observer param
    void notifyObservers(String message); // ✅ renamed + message param
    void upload(String videoTitle);     // ✅ String param
}

public class YoutubeChannel implements Subject {
    List<Observer> list = new ArrayList<>();

    @Override
    public void addObserver(Observer o) {   // ✅ Observer not String
        list.add(o);
    }

    @Override
    public void removeObserver(Observer o) { // ✅ Observer not String
        list.remove(o);
    }

    @Override
    public void upload(String videoTitle) {
        System.out.println("New video uploaded: " + videoTitle);
        notifyObservers(videoTitle);  // ✅ correct method name
    }

    @Override
    public void notifyObservers(String message) {  // ✅ renamed from update()
        for(Observer o : list) {
            o.update(message);  // ✅ passing message
        }
    }
}

// Main
public class Main {
    public static void main(String[] args) {
        YoutubeChannel yt = new YoutubeChannel();

        Observer ob1 = new EmailSubscriber("Nitesh");
        Observer ob2 = new SmsSubscriber("Ritesh");  // ✅ used SmsSubscriber too

        yt.addObserver(ob1);   // ✅ correct method name
        yt.addObserver(ob2);

        yt.upload("Video #23");  // ✅ String not int
    }
}
